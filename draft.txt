#include "FibonacciSequenceIterator.h"

FibonacciSequenceIterator::FibonacciSequenceIterator( const FibonacciSequence& aSequenceObject, uint64_t aStart ) :
    fSequenceObject(aSequenceObject),
    fIndex(aStart)
{}
    
FibonacciSequenceIterator::FibonacciSequenceIterator( uint64_t aLimit, uint64_t aStart ) :
    fSequenceObject(aLimit),
    fIndex(aStart)
{}

const uint64_t& FibonacciSequenceIterator::operator*() const
{
    return fSequenceObject.current();
}

FibonacciSequenceIterator& FibonacciSequenceIterator::operator++()
{
    fSequenceObject.advance();
    fIndex++;
    
    return *this;
}

FibonacciSequenceIterator FibonacciSequenceIterator::operator++( int )
{
    FibonacciSequenceIterator result = *this;
    
    ++(*this);
    
    return result;
}

bool FibonacciSequenceIterator::operator==( const FibonacciSequenceIterator& aOther ) const
{
    return fIndex == aOther.fIndex;
}

bool FibonacciSequenceIterator::operator!=( const FibonacciSequenceIterator& aOther) const
{
    return !(*this == aOther);
}

FibonacciSequenceIterator FibonacciSequenceIterator::begin() const
{
    FibonacciSequenceIterator Result = *this;
    
    Result.fSequenceObject.reset();
    Result.fIndex = 1;
    
    return Result;
}

FibonacciSequenceIterator FibonacciSequenceIterator::end() const
{
    FibonacciSequenceIterator Result = *this;
    uint64_t lLimit = Result.fSequenceObject.getLimit();
    
    if ( lLimit != 0 )
    {
        Result.fIndex = lLimit + 1;
    }
    else
    {
        Result.fIndex = 0;
    }

    return Result;
}

using namespace std;

FibonacciSequence::FibonacciSequence( uint64_t aLimit )
{
    fPrevious = 0;
    fCurrent = 1;
    fPosition = 1;
    fLimit = aLimit;
}

const uint64_t& FibonacciSequence::current() const
{
    return fCurrent;
}

void FibonacciSequence::advance()
{
    // infinite sequence or fLimit numbers (1..fLimit)
    if ( fLimit == 0 || fPosition <= fLimit )
    {
        uint64_t lNext = fCurrent + fPrevious;
        
        fPrevious = fCurrent;
        fCurrent = lNext;

        fPosition++;
    }
    else
    {
        throw out_of_range( "Fibonacci sequence limit exceeded" );
    }
}

const uint64_t& FibonacciSequence::getLimit() const
{
    return fLimit;
}

void FibonacciSequence::reset()
{
    fPrevious = 0;
    fCurrent = 1;
    fPosition = 1;
}

FibonacciSequenceIterator FibonacciSequence::begin() const
{
    FibonacciSequenceIterator Result( *this );
    
    return Result.begin();
}

FibonacciSequenceIterator FibonacciSequence::end() const
{
    FibonacciSequenceIterator Result( *this );
    
    return Result.end();
}

NEW HARD‑COUNTER DESIGN IDEA:
“Smart Bio‑Reinforced Slope Stabilisation System (SBRSS)”**
A cost‑effective, long‑lasting method that physically prevents landslides through biological reinforcement, water‑control, and sensor‑assisted maintenance.
This solution is widely used globally in poor mountainous regions, combining bio‑engineering + simple technology.

1. DESCRIPTION OF THE DESIGN IDEA
Instead of relying on high‑cost sensors or complex warning systems, this design uses natural slope reinforcement enhanced by smart drainage and low-cost monitoring technology.
The goal is to make slopes stronger, drier, and more stable, so landslides become much less likely.
The Smart Bio‑Reinforced Slope Stabilisation System (SBRSS) has three components:
(1) Vetiver Grass Deep-Root Barrier Lines
A cheap, natural method used in India, the Philippines, Nepal, and parts of Vietnam to stabilise slopes.
Vetiver roots grow 3–4 m deep, anchoring soil like “biological rebar.”
(2) Smart Solar Drainage Channels
Instead of expensive geotechnical drains, we create gravity-fed trenches with IoT moisture probes to keep soil dry.
(3) Smart Moisture & Flow Monitoring Points
Cheap sensors check if water is building up under the slope and notify township workers before failure happens — not for warning people to evacuate, but for maintenance (clearing drains, reinforcing areas).
This reduces landslide likelihood, not just warns about it.

2. DESIGN SPECIFICATIONS
Approach 1 — REDUCE (Reinforce Slopes)
Technology: Bio‑engineering reinforcement
Devices (Allowed in-scope):

Vetiver grass planting units (biodegradable sensor tags included)
Solar-powered moisture probes (cheap IoT)

Hardware:

Vetiver slips
Biodegradable root‑growth markers
IoT moisture probe with small solar panel
3D-printed root-depth indicator stakes (optional)

Software:

Basic soil-moisture threshold script
SMS/LoRa notification module (optional)

Function:
Deep-root vetiver rows planted every 1.5–2 m create a living hard barrier.
Moisture sensors alert the township if roots are not taking or if soil becomes too wet.

Approach 2 — ERADICATE (Eliminate Water Pressure)
Technology: Gravity-fed drainage
Devices (In-scope):

Solar drainage water‑level probes
Smart flow meter device (IoT)

Hardware:

Simple stone/soil cut-off drains
PVC‑free earth channels
Solar IoT probe
Flow sensor

Software:

Water-level threshold detection
Flow reduction alert

Function:
Channels divert rainwater away from unstable slopes.
Probes check that drains are not blocked — when water rises, township staff know to clear it.
This method prevents water buildup, which causes 90% of landslides in northern Vietnam.

Approach 3 — PREVENT (Stabilize Critical Points)
Technology: Smart root‑mesh system
Devices:

Biodegradable geotextile mesh
Low-cost tension sensor

Hardware:

Coir or jute biodegradable mesh
Tension strips
Simple pull sensors (IoT switch)

Software:

Tension-drop detection script
Local alert to maintenance team

Function:
Mesh holds topsoil in place while vegetation grows.
The tension sensor alerts if the mesh tears or shifts, signalling slope weakening.

3. HOW THE DESIGN WORKS
STEP 1 — Plant Deep-Root Vetiver Rows
Vetiver grows downward like a “biological nail.”
These roots:

Interlock soil
Stabilise steep slopes
Reduce erosion
Resist floods, drought, and livestock movement

STEP 2 — Carve Simple Drainage Trenches
Cheap, labour-based channels built by community volunteers remove groundwater pressure.
STEP 3 — Install Micro IoT Sensors
Sensors do NOT warn villagers — they warn maintenance teams.
If water builds up or mesh weakens → township teams fix it.
STEP 4 — Long-term strengthening
Within 6–12 months:

Vetiver completely binds the slope
Mesh biodegrades after soil stabilises
Maintenance becomes near-zero
